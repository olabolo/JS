var a = 1, b = 1, c, d;
c = ++a; alert(c);           // 2 префиксная форма a = 1+1 = 2, c = 2
d = b++; alert(d);           // 1 постфиксная форма b = 1+1 = 2, d = 1
c = (2+ ++a); alert(c);      // 5 c = (2+1+2) a = 3
d = (2+ b++); alert(d);      // 4 d = (2+2) b = 2+1 = 3
alert(a);                    // 3
alert(b);                    // 3
